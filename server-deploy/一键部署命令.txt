## 第一步：安装 Docker（粘贴执行）

apt update && apt install -y docker.io && systemctl start docker && systemctl enable docker

## 第二步：部署服务（粘贴执行）

mkdir -p ~/toremote-server/data && cat > ~/toremote-server/docker-compose.yml << 'EOF'
version: '3'
services:
  hbbs:
    container_name: hbbs
    image: rustdesk/rustdesk-server:latest
    command: hbbs
    volumes:
      - ./data:/root
    network_mode: host
    restart: unless-stopped
  hbbr:
    container_name: hbbr
    image: rustdesk/rustdesk-server:latest
    command: hbbr
    volumes:
      - ./data:/root
    network_mode: host
    restart: unless-stopped
EOF
cd ~/toremote-server && docker compose up -d && sleep 3 && echo "=== 部署完成 ===" && echo "公钥:" && cat ~/toremote-server/data/id_ed25519.pub
